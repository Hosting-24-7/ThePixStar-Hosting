async function loginAdmin() { const pass = document.getElementById('adminPass').value; const response = await fetch('http://localhost:5000/api/admin/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ password: pass }) }); const data = await response.json(); if (data.success) { // Admin session set karein ya direct dashboard par bhejen localStorage.setItem('isAdmin', 'true'); window.location.href = "admin_dashboard.html"; } else { alert("Invalid Password! Access Denied."); } }
